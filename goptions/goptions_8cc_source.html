128 // returns a vector of json objects of the base jcard plus all imported jcards
129 vector<json> GOptions::getUserJsonsFromJCard(string jcardFilename)
130 {
131  vector<json> userJsons;
132 
133  // nothing happens if jcard wasn't found
134  if (jcardFilename == UNINITIALIZEDSTRINGQUANTITY) {
135  return userJsons;
136  }
137 
138  // base jcard
139  // removing '#' from "base" (command line) jcard
140  // function is defined in gutilities
141  string basePureJsonString = parseFileAndRemoveComments(jcardFilename, "#");
142 
143  if ( gdebug ) {
144  cout << endl << GREENSQUAREITEM << " Parsing base jcard content: " << endl << basePureJsonString << endl;
145  }
146 
147  // building json object from base jcard
148  json baseJson = json::parse(basePureJsonString);
149 
150  // now imports imported jcards, if any
151  for (auto& [key, value] : baseJson.items()) {
152  if ( key == IMPORTJCARD ) {
153 
154  // value is a vector<string>
155  for ( auto &importedJCardName: value) {
156  // adding extension to set fileName
157  string importJcardFileName = replaceCharInStringWithChars(importedJCardName, "\"", "") + ".jcard";
158 
159  // import json (this exit if filename isn't there)
160  string importedParsedJson = parseFileAndRemoveComments(importJcardFileName);
161 
162  // add imported json to userJsons vector
163  userJsons.push_back(json::parse(importedParsedJson));
164  }
165  }
166  }
167 
168  // appending the base jcard json at the end:
169  // all imports should be declared at the top of the jcard thus they come before the base settings
170  userJsons.push_back(baseJson);
171  return userJsons;

175 // parse all the jsons from the jcards / command line in the GOptions array
176 void GOptions::parseJSONSIntoGOption(vector<json> allUserJsons)
177 {
178  // looping over all jsons
179  for (auto& userJsonOption: allUserJsons) {
180 
181  // looping over all json inside each userJsonOption
182  for (auto& [userJsonKey, userJsonValue] : userJsonOption.items()) {
183 
184  // skipping import directives, the imported json are already here
185  if (userJsonKey == IMPORTJCARD) {
186  continue;
187  }
188 
189  // if the first character is "+", this is an addition
190  bool isAnAddition = ( userJsonKey.front() == '+' );
191 
192  string userJsonKeyRoot = userJsonKey;
193 
194  // if it's an addition, remove first char
195  if ( isAnAddition ) {
196  userJsonKeyRoot = userJsonKey.substr(1, userJsonKey.size() - 1);
197  }
198 
199  if ( gdebug ) {
200  cout << endl << GREENSQUAREITEM << "Looking to assign Json Key " << BOLDWHHL << userJsonKey << RSTHHR << endl;
201  cout << GTAB << "Content: " << userJsonValue << endl;
202  cout << GTAB << "userJsonKeyRoot: " << userJsonKeyRoot << endl;
203  cout << GTAB << "Is an addition: " << isAnAddition << endl;
204  }
205 
206  // GOption index, UNINITIALIZEDNUMBERQUANTITY if not found
207  long userJsonOptionDefinitionIndex = findOptionIndex(userJsonKeyRoot);
208 
209  if ( gdebug ) {
210  cout << GREENSQUAREITEM << "Option " << BOLDWHHL << userJsonKeyRoot << RSTHHR << " definition found." << endl;
211  }
212 
213  goptions.at(userJsonOptionDefinitionIndex).assignValuesFromJson(userJsonKey, userJsonValue, isAnAddition, gdebug, gstrict);
214  }
215  }
