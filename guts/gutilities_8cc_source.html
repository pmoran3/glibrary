131 double gutilities::getG4Number(string v, bool warnIfNotUnit)
132 {
133  string value = trimSpacesFromString(v);
134 
135  // no * found
136  if(value.find("*") == string::npos) {
137  // no * found, this should be a number
138  // No unit is still ok if the number is 0
139 
140  if(value.length()>0 && warnIfNotUnit && stod(value) != 0) {
141  cerr << " ! Warning: value " << v << " does not contain units." << endl;
142  }
143 
144  double answer = 0;
145 
146  try {
147  answer = stod(value);
148  }
149  catch(exception& e) {
150  cerr << FATALERRORL << "stod exception in gutilities: could not convert string to double " << e.what() << endl;
151  }
152 
153  return answer;
154 
155  } else {
156  string rootValue = value.substr(0, value.find("*"));
157  string units = value.substr(value.find("*") + 1);
158 
159  double answer;
160 
161  try {
162  answer = stod(rootValue);
163  }
164  catch(exception& e) {
165  cerr << FATALERRORL << "stod exception in gutilities: could not convert string to double " << e.what() << endl;
166  }
167 
168 
169  if( units == "m") answer *= m;
170  else if( units == "inches") answer *= 2.54*cm;
171  else if( units == "inch") answer *= 2.54*cm;
172  else if( units == "cm") answer *= cm;
173  else if( units == "mm") answer *= mm;
174  else if( units == "um") answer *= 1E-6*m;
175  else if( units == "fm") answer *= 1E-15*m;
176  else if( units == "deg") answer *= deg;
177  else if( units == "degrees") answer *= deg;
178  else if( units == "arcmin") answer = answer/60.0*deg;
179  else if( units == "rad") answer *= rad;
180  else if( units == "mrad") answer *= mrad;
181  else if( units == "eV") answer *= eV;
182  else if( units == "MeV") answer *= MeV;
183  else if( units == "KeV") answer *= 0.001*MeV;
184  else if( units == "GeV") answer *= GeV;
185  else if( units == "T") answer *= tesla;
186  else if( units == "T/m") answer *= tesla/m;
187  else if( units == "Tesla") answer *= tesla;
188  else if( units == "gauss") answer *= gauss;
189  else if( units == "kilogauss") answer *= gauss*1000;
190  else if( units == "ns") answer *= ns;
191  else if( units == "ms") answer *= ms;
192  else if( units == "us") answer *= us;
193  else if( units == "counts") answer *= 1;
194  else cerr << GWARNING << ">" << units << "<: unit not recognized for string <" << v << ">" << endl;
195  return answer;
196  }
197 
198  return EXIT_SUCCESS;
199 }
